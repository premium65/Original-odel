### 24-Hour Cooldown Calculation
```javascript
// Check if ad is available:
const lastClickTime = new Date(ad.lastClickedAt);
const now = new Date();
const hoursSinceClick = (now - lastClickTime) / (1000 * 60 * 60);

if (hoursSinceClick >= 24) {
  // Ad is AVAILABLE - can click
  return { available: true };
} else {
  // Ad is LOCKED - show countdown
  const timeLeft = 24 - hoursSinceClick;
  const hours = Math.floor(timeLeft);
  const minutes = Math.floor((timeLeft - hours) * 60);
  const seconds = Math.floor(((timeLeft - hours) * 60 - minutes) * 60);
  return { available: false, countdown: `${hours}:${minutes}:${seconds}` };
}
```
